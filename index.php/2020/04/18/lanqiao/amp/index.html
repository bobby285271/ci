问长度为 8 的有效括号序列。把所有情况枚举了一遍，在任意位置右括号多于左括号就叉掉，最后看左右括号是否相等。

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll ans = 0;
int main()
{
    for (int i = 0; i < pow(2, 8); i++)
    {
        int a[8];
        int temp = i;
        for (int j = 0; j < 8; j++)
        {
            a[j] = temp % 2;
            temp /= 2;
        }
        int check = 0;
        for (int j = 0; j < 8; j++)
        {
            // cout << check;
            if (a[j] == 0)
                check++;
            else
                check--;
            if (check < 0)
            {
                check = -1;
                break;
            }
        }
        if (check == 0)
        {
            for (int j = 0; j < 8; j++)
            {
                if (a[j] == 0)
                    cout << "(";
                else
                    cout << ")";
            }
            cout << endl;
            ans++;
        }
    }
    cout << ans << endl;
}

ans = 14

问某个区间内不能被 a,b 和 c 整除的数，容斥一下就可。应该不存在 TLE 数据，WA 不知道，欢迎 hack

#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
typedef long long ll;

ll gcd(ll a, ll b) { return b == 0 ? a : gcd(b, a % b); }
ll lcm(ll a, ll b) { return a * b / gcd(a, b); }

int main()
{
    ll n, a, b, c, ans;
    cin >> n >> a >> b >> c;
    ans = n - n / a - n / b - n / c;
    ans += n / lcm(a, b) + n / lcm(c, b) + n / lcm(c, a);
    ans -= n / lcm(a, lcm(b, c));
    cout << ans << endl;
    return 0;
}

    if (c > m / 2)
        tempc = m + 1 - c;
    if (r > n / 2)
        tempr = n + 1 - r;

    while (cur != circle)
    {
        ans += (n - 2 * cur + 1) * 2 + (m - 2 * cur + 1) * 2;

    ll tom = m - 2 * circle + 1;
    // cout << circle << endl;
    if (r != circle)
        ans += tom;
    else
    {
        cout << ans + c - circle << endl;

    }

要求一个数列偶数项要小于相邻的两个奇数项。直接暴力，目测是 TLE 做法，正解不会

#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
typedef long long ll;

ll gcd(ll a, ll b) { return b == 0 ? a : gcd(b, a % b); }
ll lcm(ll a, ll b) { return a * b / gcd(a, b); }

ll ans = 0;
ll n, m;

void dfs(ll curn, ll index)
{
    if (index == m - 1)
    {
        if (index % 2 == 0)
        {
            ans += n - curn;
            ans %= 10000;
        }
        else
        {
            ans += curn;
            ans %= 10000;
        }
    }
    else
    {
        if (index % 2 == 0)
        {
            for (ll i = curn + 1; i <= n; i++)
            {
                dfs(i, index + 1);
            }
        }
        else
        {
            for (ll i = 1; i < curn; i++)
            {
                dfs(i, index + 1);
            }
        }
    }
}

int main()
{
    cin >> m >> n;
    if (m == 1)
    {
        cout << n % 10000 << endl;
        return 0;
    }
    if (m == 2)
    {
        cout << (n * (n - 1) / 2) % 10000 << endl;
        return 0;
    }
    for (ll i = 1; i <= n; i++)
    {
        dfs(i, 1);
    }
    cout << ans << endl;
    return 0;
}

            a[j][i] = a[i][j];
        }

    for (int i = 1; i <= n; i++)

        minn = 100000000;
        for (int j = 1; j <= n; j++)
        {
            if (vis[j] == 0 && dis[j] < minn)

                minn = dis[j];

给出多个坐标和半径，对于某个点可以画一个对应半径的圆，也可以不画，要求画出来的圆不能有重叠。问面积最大值。暴力... TLE 做法，正解不会

#include <iostream>
#include <cmath>
using namespace std;
typedef long long ll;
int main()
{
    ll n;
    cin >> n;
    ll x[n + 10], y[n + 10], r[n + 10];
    bool p[n + 10];
    for (ll i = 0; i < n; i++)
    {
        cin >> x[i] >> y[i] >> r[i];
    }
    ll ans = 0;
    for (ll i = 0; i < (1 << n); i++)
    {
        ll tempans = 0;
        ll temp = i;
        for (ll j = 0; j < n; j++)
        {
            p[j] = temp % 2;
            temp /= 2;
        }
        bool flag = 0;
        for (ll i = 0; i < n; i++)
        {
            if (p[i] == 1)
                tempans += r[i] * r[i];
            for (ll j = i + 1; j < n; j++)
            {
                if (p[i] == 1 && p[j] == 1)
                {
                    if ((x[i] - x[j]) * (x[i] - x[j]) + (y[i] - y[j]) * (y[i] - y[j]) < (r[i] + r[j]) * (r[i] + r[j]))
                    {
                        flag = 1;
                        break;
                    }
                }
            }
            if (flag == 1)
                break;
        }
        if (flag == 0)
        {
            ans = max(ans, tempans);
            // cout << ans << endl;
        }
    }
    cout << ans << endl;
}
